# 函数模板语法

```c++
#include <iostream>

using namespace std;

// 声明一个模版，告诉编译器后边代码中的T不要报错，T是一个通用的数据类型。
template <typename T>

void mySwap(T &a, T &b)
{
    T temp = a;
    a = b;
    b = temp;
}

void test01()
{
    int a = 10;
    int b = 200;
    // 模板使用方法
    // 1, 自动类型推导
    mySwap(a, b);
    cout << a << ", " << b << endl;

    // 2, 显式制定类型
    mySwap<int>(a, b);
    cout << a << ", " << b << endl;
}

int main(int argc, char const *argv[])
{

    test01();
    return 0;
}
```
